014    /**
015     * Abstract base class for all statement types
016     * that support parameters, i.e. <code>PreparedStatement</code>
017     * and <code>CallableStatement</code>.
018     */
019    public abstract class AbstractParameterResultSetHandler extends AbstractResultSetHandler

023        private Map updateCountForStatement = new HashMap();
024        private Map throwsSQLException = new HashMap();

027            /**
028             * Collects all SQL strings that were executed.
029             * @param sql the SQL string
030             * @param parameters a copy of the corresponding parameter map
031             */
032            public void addParameterMapForExecutedStatement(String sql, Map parameters)
033            {
034                    if(null != parameters)
035                    {
036                            if(null == executedStatementParameters.get(sql))
037                            {
038                                    executedStatementParameters.put(sql, new ParameterSets(sql));
039                            }
040                            ParameterSets sets = (ParameterSets)executedStatementParameters.get(sql);
041                            sets.addParameterSet(parameters);
042                    }
043            }
044            
045            /**
046             * Returns the <code>ParameterSets</code> for a specified
047             * SQL string.
048             * @param sql the SQL string
049             * @return the <code>Map</code> of parameters
050             */
051            public ParameterSets getParametersForExecutedStatement(String sql)
052            {
053                    return (ParameterSets)executedStatementParameters.get(sql);
054            }
055            
056            /**
057             * Returns the <code>Map</code> of executed SQL strings.
058             * Each string maps to the corresponding {@link ParameterSets}
059             * object.
060             * @return the <code>Map</code> of parameters
061             */
062            public Map getExecutedStatementParameter()
063            {
064                    return Collections.unmodifiableMap(executedStatementParameters);
065            }

081        /**
082         * Returns the first update count that matches the
083         * specified SQL string and the specified parameters. 
084         * Please note that you can modify the match parameters with 
085         * {@link #setCaseSensitive}, {@link #setExactMatch} and 
086         * {@link #setUseRegularExpressions} and the match parameters for the 
087         * specified parameter list with {@link #setExactMatchParameter}.
088         * @param sql the SQL string
089         * @param parameters the parameters
090         * @return the corresponding update count
091         */
092        public Integer getUpdateCount(String sql, Map parameters)
093        {
094            SQLStatementMatcher matcher = new SQLStatementMatcher(getCaseSensitive(), getExactMatch(), getUseRegularExpressions());
095            List list = matcher.getMatchingObjects(updateCountForStatement, sql, true, true);
096            for(int ii = 0; ii < list.size(); ii++)
097            {
098                MockUpdateCountWrapper wrapper = (MockUpdateCountWrapper)list.get(ii);
099                if(doParameterMatch(wrapper.getParamters(), parameters))
100                {
101                    return wrapper.getUpdateCount();
102                }
103            }
104            return null;
105        }

151        /**
152         * Returns the <code>SQLException</code> the specified SQL string
153         * should throw. Returns <code>null</code> if the specified SQL string
154         * should not throw an exception.
155         * This can be used to simulate database exceptions.
156         * Please note that you can modify the match parameters with 
157         * {@link #setCaseSensitive}, {@link #setExactMatch} and 
158         * {@link #setUseRegularExpressions} and the match parameters for the 
159         * specified parameter list with {@link #setExactMatchParameter}.
160         * @param sql the SQL string
161         * @param parameters the parameters
162         * @return the <code>SQLException</code> or <code>null</code>
163         */
164        public SQLException getSQLException(String sql, Map parameters)
165        {
166            SQLStatementMatcher matcher = new SQLStatementMatcher(getCaseSensitive(), getExactMatch(), getUseRegularExpressions());
167            List list = matcher.getMatchingObjects(throwsSQLException, sql, true, true);
168            for(int ii = 0; ii < list.size(); ii++)
169            {
170                MockSQLExceptionWrapper wrapper = (MockSQLExceptionWrapper)list.get(ii);
171                if(doParameterMatch(wrapper.getParamters(), parameters))
172                {
173                    return wrapper.getException();
174                }
175            }
176            return null;
177        }

179        protected boolean doParameterMatch(Map expectedParameters, Map actualParameters)
180        {
181            if(exactMatchParameter)
182            {
183                if(actualParameters.size() != expectedParameters.size()) return false;
184                Iterator iterator = actualParameters.keySet().iterator();
185                while(iterator.hasNext())
186                {
187                    Object currentKey = iterator.next();
188                    if(!actualParameters.containsKey(currentKey)) return false;
189                    Object expectedObject = expectedParameters.get(currentKey);
190                    if(!ParameterUtil.compareParameter(actualParameters.get(currentKey), expectedObject))
191                    {
192                        return false;
193                    }
194                }
195                return true;
196            }
197            else
198            {
199                Iterator iterator = expectedParameters.keySet().iterator();
200                while(iterator.hasNext())
201                {
202                    Object currentKey = iterator.next();
203                    if(!actualParameters.containsKey(currentKey)) return false;
204                    Object actualObject = actualParameters.get(currentKey);
205                    if(!ParameterUtil.compareParameter(actualObject, expectedParameters.get(currentKey)))
206                    {
207                        return false;
208                    }
209                }
210                return true;
211            }
212        }

263         * Prepare a <code>ResultSet</code> for a specified SQL string and
264         * the specified parameters. The specified parameters <code>List</code>
265         * must contain the parameters in the correct order starting with index 0 for
266         * the first parameter. Please keep in mind that parameters in
267         * <code>PreparedStatement</code> objects start with 1 as the first
268         * parameter. So <code>parameters.get(0)</code> maps to the
269         * parameter with index 1.

276         * @param parameters the parameters

279        {
280            Map params = new HashMap();
281            for(int ii = 0; ii < parameters.size(); ii++)

283                params.put(new Integer(ii + 1), parameters.get(ii));
284            }
285            prepareResultSet(sql, resultSet, params);
286        }

310            }
311            list.add(new MockResultSetWrapper(resultSet, new HashMap(parameters)));
312        }
