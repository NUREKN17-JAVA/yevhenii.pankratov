103        public ResultSet executeQuery(String sql) throws SQLException
104        {
105            SQLException exception = resultSetHandler.getSQLException(sql);
106            if(null != exception)
107            {
108                throw exception;
109            }
110            resultSetHandler.addExecutedStatement(sql);
111            MockResultSet result = resultSetHandler.getResultSet(sql);
112            if(null != result)
113            {
114                result = cloneResultSet(result);
115                resultSetHandler.addReturnedResultSet(result);
116                setResultSet(result);
117                return result;
118            }
119            result = cloneResultSet(resultSetHandler.getGlobalResultSet());
120            resultSetHandler.addReturnedResultSet(result);
121            setResultSet(result);
122            return result;
123        }
124    
125        public int executeUpdate(String sql) throws SQLException
126        {
127            SQLException exception = resultSetHandler.getSQLException(sql);
128            if(null != exception)
129            {
130                throw exception;
131            }
132            resultSetHandler.addExecutedStatement(sql);
133            Integer returnValue = resultSetHandler.getUpdateCount(sql);
134            if(null != returnValue)
135            {
136                int updateCount = returnValue.intValue();
137                setUpdateCount(updateCount);
138                return updateCount;
139            }
140            int updateCount = resultSetHandler.getGlobalUpdateCount();
141            setUpdateCount(updateCount);
142            return updateCount;
143        }
144        
145        public boolean execute(String sql) throws SQLException
146        {
147            boolean callExecuteQuery = isQuery(sql);
148            if(callExecuteQuery)
149            {
150                executeQuery(sql);
151            }
152            else
153            {
154                executeUpdate(sql);
155            }
156            return callExecuteQuery;
157        }
158        
159        public int[] executeBatch() throws SQLException
160        {
161            int[] results = new int[batches.size()];
162            for(int ii = 0; ii < results.length; ii++)
163            {
164                String nextSQL = (String)batches.get(ii);
165                if(isQuery(nextSQL))
166                {
167                    throw new BatchUpdateException("SQL " + batches.get(ii) + " in the list of batches returned a ResultSet.", null);
168                }
169                results[ii] = executeUpdate(nextSQL);
170            }
171            return results;
172        }
173        
174        public int executeUpdate(String sql, int autoGeneratedKeys) throws SQLException
175        {
176            int updateCount = executeUpdate(sql);
177            setGeneratedKeysResultSet(sql, autoGeneratedKeys);
178            return updateCount;
179        }

208        private void setGeneratedKeysResultSet(String sql, int autoGeneratedKeys) throws SQLException
209        {
210            if(Statement.RETURN_GENERATED_KEYS != autoGeneratedKeys && Statement.NO_GENERATED_KEYS != autoGeneratedKeys)
211            {
212                throw new SQLException("autoGeneratedKeys must be either Statement.RETURN_GENERATED_KEYS or Statement.NO_GENERATED_KEYS");
213            }
214            if(Statement.RETURN_GENERATED_KEYS == autoGeneratedKeys)
215            {
216                lastGeneratedKeys = resultSetHandler.getGeneratedKeys(sql);
217                if(null == lastGeneratedKeys)
218                {
219                    lastGeneratedKeys = resultSetHandler.getGlobalGeneratedKeys();
220                }
221            }
222            else
223            {
224                lastGeneratedKeys = null;
225            }
226        }

288        protected boolean isQuery(String sql)
289        {
290            boolean isQuery;
291            Boolean returnsResultSet = resultSetHandler.getReturnsResultSet(sql);
292            if(null != returnsResultSet)
293            {
294                isQuery = returnsResultSet.booleanValue();
295            }
296            else
297            {
298                isQuery = SQLUtil.isSelect(sql);
299            }
300            return isQuery;
301        }

321            return false;
